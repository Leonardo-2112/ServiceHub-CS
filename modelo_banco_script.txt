    ALGORITHM = UNDEFINED 
VIEW `vw_produto` AS
    SELECT 
        `p`.`id` AS `id`,
        `p`.`cod_barras` AS `cod_barras`,
        `p`.`descricao` AS `descricao`,
        `p`.`categoria_id` AS `categoria_id`,
        `c`.`nome` AS `categoria`,
        `e`.`quantidade` AS `quantidade`,
        `p`.`valor_unit` AS `valor_unit`,
        `p`.`unidade_venda` AS `unidade_venda`,
        `p`.`estoque_minimo` AS `estoque_minimo`,
        `p`.`classe_desconto` AS `classe_desconto`,
        `p`.`imagem` AS `imagem`,
        `p`.`data_cad` AS `data_cad`,
        `p`.`descontinuado` AS `descontinuado`,
        `e`.`data_ultimo_movimento` AS `data_ultimo_movimento`
    FROM
        ((`produtos` `p`
        JOIN `categorias` `c` ON (`p`.`categoria_id` = `c`.`id`))
        JOIN `estoques` `e` ON (`p`.`id` = `e`.`produto_id`))
    ORDER BY `p`.`descricao`
